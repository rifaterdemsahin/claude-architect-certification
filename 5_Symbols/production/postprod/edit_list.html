🛠️ SQL Setup Instructions 
-- 1. Main Video Registry
CREATE TABLE IF NOT EXISTS public.certification_videos (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL,
    module_number INT NOT NULL DEFAULT 1,
    video_number INT NOT NULL DEFAULT 1,
    title TEXT NOT NULL,
    description TEXT,
    edit_todo TEXT,
    embed_url TEXT NOT NULL,
    share_url TEXT NOT NULL
);

-- 2. Granular Video Assets (Research, Artifacts, Sentences, Shots)
CREATE TABLE IF NOT EXISTS public.video_assets (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL,
    video_id BIGINT REFERENCES public.certification_videos(id) ON DELETE CASCADE,
    asset_type TEXT NOT NULL, -- research, artifact, sentence, shot
    content TEXT NOT NULL,
    is_used BOOLEAN DEFAULT false
);

-- Enable RLS
ALTER TABLE public.certification_videos ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.video_assets ENABLE ROW LEVEL SECURITY;

-- Allow public all access
CREATE POLICY "Allow public all access" ON public.certification_videos FOR ALL USING (true);
CREATE POLICY "Allow public all access" ON public.video_assets FOR ALL USING (true);
